Appendix: Useful formulas¶

Finite difference operator notation¶

\[\tag{393} u'(t_n) \approx \lbrack D_tu\rbrack^n = \frac{u^{n+\frac{1}{2}} - u^{n-\frac{1}{2}}}{\Delta t}\]
\[\tag{394} u'(t_n) a\approx \lbrack D_{2t}u\rbrack^n = \frac{u^{n+1} - u^{n-1}}{2\Delta t} +\]
\[\tag{395} u'(t_n) = \lbrack D_t^-u\rbrack^n = \frac{u^{n} - u^{n-1}}{\Delta t}\]
\[\tag{396} u'(t_n) \approx \lbrack D_t^+u\rbrack^n = \frac{u^{n+1} - u^{n}}{\Delta t}\]
\[\tag{397} u'(t_{n+\theta}) = \lbrack \bar D_tu\rbrack^{n+\theta} = \frac{u^{n+1} - u^{n}}{\Delta t}\]
\[\tag{398} u'(t_n) \approx \lbrack D_t^{2-}u\rbrack^n = \frac{3u^{n} - 4u^{n-1} + u^{n-2}}{2\Delta t}\]
\[\tag{399} u''(t_n) \approx \lbrack D_tD_t u\rbrack^n = \frac{u^{n+1} - 2u^{n} + u^{n-1}}{\Delta t^2}\]
\[\tag{400} u(t_{n+\frac{1}{2}}) \approx \lbrack \overline{u}^{t}\rbrack^{n+\frac{1}{2}} = \frac{1}{2}(u^{n+1} + u^n)\]
\[\tag{401} u(t_{n+\frac{1}{2}})^2 \approx \lbrack \overline{u^2}^{t,g}\rbrack^{n+\frac{1}{2}} = u^{n+1}u^n\]
\[\tag{402} u(t_{n+\frac{1}{2}}) \approx \lbrack \overline{u}^{t,h}\rbrack^{n+\frac{1}{2}} = \frac{2}{\frac{1}{u^{n+1}} + \frac{1}{u^n}}\]
\[\tag{403} u(t_{n+\theta}) \approx \lbrack \overline{u}^{t,\theta}\rbrack^{n+\theta} = \theta u^{n+1} + (1-\theta)u^n ,\quad t_{n+\theta}=\theta t_{n+1} + (1-\theta)t_{n-1}\]

Truncation errors of finite difference approximations¶

\[{u_{\small\mbox{e}}}'(t_n) = [D_t{u_{\small\mbox{e}}}]^n + R^n = \frac{{u_{\small\mbox{e}}}^{n+\frac{1}{2}} - {u_{\small\mbox{e}}}^{n-\frac{1}{2}}}{\Delta t} +R^n\nonumber,\]
\[\tag{404} R^n = -\frac{1}{24}{u_{\small\mbox{e}}}'''(t_n)\Delta t^2 + {\cal O}(\Delta t^4)\]
\[{u_{\small\mbox{e}}}'(t_n) = [D_{2t}{u_{\small\mbox{e}}}]^n +R^n = \frac{{u_{\small\mbox{e}}}^{n+1} - {u_{\small\mbox{e}}}^{n-1}}{2\Delta t} + R^n\nonumber,\]
\[\tag{405} R^n = -\frac{1}{6}{u_{\small\mbox{e}}}'''(t_n)\Delta t^2 + {\cal O}(\Delta t^4)\]
\[{u_{\small\mbox{e}}}'(t_n) = [D_t^-{u_{\small\mbox{e}}}]^n +R^n = \frac{{u_{\small\mbox{e}}}^{n} - {u_{\small\mbox{e}}}^{n-1}}{\Delta t} +R^n\nonumber,\]
\[\tag{406} R^n = -\frac{1}{2}{u_{\small\mbox{e}}}''(t_n)\Delta t + {\cal O}(\Delta t^2)\]
\[{u_{\small\mbox{e}}}'(t_n) = [D_t^+{u_{\small\mbox{e}}}]^n +R^n = \frac{{u_{\small\mbox{e}}}^{n+1} - {u_{\small\mbox{e}}}^{n}}{\Delta t} +R^n\nonumber,\]
\[\tag{407} R^n = -\frac{1}{2}{u_{\small\mbox{e}}}''(t_n)\Delta t + {\cal O}(\Delta t^2)\]
\[{u_{\small\mbox{e}}}'(t_{n+\theta}) = [\bar D_t{u_{\small\mbox{e}}}]^{n+\theta} +R^{n+\theta} = \frac{{u_{\small\mbox{e}}}^{n+1} - {u_{\small\mbox{e}}}^{n}}{\Delta t} +R^{n+\theta}\nonumber,\]
\[R^{n+\theta} = -\frac{1}{2}(1-2\theta){u_{\small\mbox{e}}}''(t_{n+\theta})\Delta t + \frac{1}{6}((1 - \theta)^3 - \theta^3){u_{\small\mbox{e}}}'''(t_{n+\theta})\Delta t^2 + \nonumber\]
\[\tag{408} \quad {\cal O}(\Delta t^3)\]
\[{u_{\small\mbox{e}}}'(t_n) = [D_t^{2-}{u_{\small\mbox{e}}}]^n +R^n = \frac{3{u_{\small\mbox{e}}}^{n} - 4{u_{\small\mbox{e}}}^{n-1} + {u_{\small\mbox{e}}}^{n-2}}{2\Delta t} +R^n\nonumber,\]
\[\tag{409} R^n = \frac{1}{3}{u_{\small\mbox{e}}}'''(t_n)\Delta t^2 + {\cal O}(\Delta t^3)\]
\[{u_{\small\mbox{e}}}''(t_n) = [D_tD_t {u_{\small\mbox{e}}}]^n +R^n = \frac{{u_{\small\mbox{e}}}^{n+1} - 2{u_{\small\mbox{e}}}^{n} + {u_{\small\mbox{e}}}^{n-1}}{\Delta t^2} +R^n\nonumber,\]
\[\tag{410} R^n = -\frac{1}{12}{u_{\small\mbox{e}}}''''(t_n)\Delta t^2 + {\cal O}(\Delta t^4)\]
\[{u_{\small\mbox{e}}}(t_{n+\theta}) = [\overline{{u_{\small\mbox{e}}}}^{t,\theta}]^{n+\theta} +R^{n+\theta} = \theta {u_{\small\mbox{e}}}^{n+1} + (1-\theta){u_{\small\mbox{e}}}^n +R^{n+\theta},\nonumber\]
\[\tag{411} R^{n+\theta} = -\frac{1}{2}{u_{\small\mbox{e}}}''(t_{n+\theta})\Delta t^2\theta (1-\theta) + {\cal O}(\Delta t^3) {\thinspace .}\]

Finite differences of exponential functions¶

Complex exponentials¶

Let \(u^n = \exp{(i\omega n\Delta t)} = e^{i\omega t}\).

\[\tag{412} [D_tD_t u]^n = u^n \frac{2}{\Delta t}(\cos \omega\Delta t - 1) = -\frac{4}{\Delta t}\sin^2\left(\frac{\omega\Delta t}{2}\right),\]
\[\tag{413} [D_t^+ u]^n = u^n\frac{1}{\Delta t}(\exp{(i\omega\Delta t)} - 1),\]
\[\tag{414} [D_t^- u]^n = u^n\frac{1}{\Delta t}(1 - \exp{(-i\omega\Delta t)}),\]
\[\tag{415} [D_t u]^n = u^n\frac{2}{\Delta t}i\sin{\left(\frac{\omega\Delta t}{2}\right)},\]
\[\tag{416} [D_{2t} u]^n = u^n\frac{1}{\Delta t}i\sin{(\omega\Delta t)}\]\[ {\thinspace .}\]

Real exponentials¶

Let \(u^n = \exp{(\omega n\Delta t)} = e^{\omega t}\).

\[\tag{417} [D_tD_t u]^n = u^n \frac{2}{\Delta t}(\cos \omega\Delta t - 1) = -\frac{4}{\Delta t}\sin^2\left(\frac{\omega\Delta t}{2}\right),\]
\[\tag{418} [D_t^+ u]^n = u^n\frac{1}{\Delta t}(\exp{(i\omega\Delta t)} - 1),\]
\[\tag{419} [D_t^- u]^n = u^n\frac{1}{\Delta t}(1 - \exp{(-i\omega\Delta t)}),\]
\[\tag{420} [D_t u]^n = u^n\frac{2}{\Delta t}i\sin{\left(\frac{\omega\Delta t}{2}\right)},\]
\[\tag{421} [D_{2t} u]^n = u^n\frac{1}{\Delta t}i\sin{(\omega\Delta t)}\]\[ {\thinspace .}\]

Finite differences of \(t^n\)¶

The following results are useful when checking if a polynomial term in a solution fulfills the discrete equation for the numerical method.

\[\tag{422} \lbrack D_t^+ t\rbrack^n = 1,\]
\[\tag{423} \lbrack D_t^- t\rbrack^n = 1,\]
\[\tag{424} \lbrack D_t t\rbrack^n = 1,\]
\[\tag{425} \lbrack D_{2t} t\rbrack^n = 1,\]
\[\tag{426} \lbrack D_{t}D_t t\rbrack^n = 0\]\[ {\thinspace .}\]

The next formulas concern the action of difference operators on a \(t^2\) term.

\[\tag{427} \lbrack D_t^+ t^2\rbrack^n = (2n+1)\Delta t,\]
\[\tag{428} \lbrack D_t^- t^2\rbrack^n = (2n-1)\Delta t,\]
\[\tag{429} \lbrack D_t t^2\rbrack^n = 2n\Delta t,\]
\[\tag{430} \lbrack D_{2t} t^2\rbrack^n = 2n\Delta t,\]
\[\tag{431} \lbrack D_{t}D_t t^2\rbrack^n = 2,\]

Finally, we present formulas for a \(t^3\) term: These must be controlled against lib.py. Use \(t_n\) instead of \(n\Delta t\)??

\[\tag{432} \lbrack D_t^+ t^3\rbrack^n = 3(n\Delta t)^2 + 3n\Delta t^2 + \Delta t^2,\]
\[\tag{433} \lbrack D_t^- t^3\rbrack^n = 3(n\Delta t)^2 - 3n\Delta t^2 + \Delta t^2,\]
\[\tag{434} \lbrack D_t t^3\rbrack^n = 3(n\Delta t)^2 + \frac{1}{4}\Delta t^2,\]
\[\tag{435} \lbrack D_{2t} t^3\rbrack^n = 3(n\Delta t)^2 + \Delta t^2,\]
\[\tag{436} \lbrack D_{t}D_t t^3\rbrack^n = 6n\Delta t,\]
